Métodos para crear un álbum en Cloud Firestore

En este apartado se explica los métodos implementados para crear un álbum en las secciones: servicio, componente y vista.

Los métodos implementados a continuación son propios de Firebase - Cloud Firestore y Storage

Metodos en el Servicio

Método para guardar imagen en Cloud Firestorage y obtener el link de descarga.

            
            import { AngularFirestore , AngularFirestoreCollection} from '@angular/fire/compat/firestore';
            import { AngularFireStorage } from "@angular/fire/compat/storage";
            //Crea una referencia a Firestore con los campos de álbum
            albumCollection: AngularFirestoreCollection<Album>;
            constructor(private angularFirestore: AngularFirestore,private storage: AngularFireStorage) {
                //Especifica colección en Firestore
                this.albumCollection= angularFirestore.collection("albums");
            }
            //Método para subir archivo a Firebase Storage y obtener la URL 
            addGenreAlbum(object: any, _file: any, collection: string) {
                //cambio de ruta dependiendo de la colección
                const filePath = collection+ "/" + object.name; //en caso que sea albumes
            }
            //Creación de la referencia en FireStorage dependiendo de la ruta
                const ref = this.storage.ref(this.filePath); 
                
                //Agregar el archivo dependiendo de la referencia
                ref.put(_file).then(() => {
                //obtiene la URL del archivo subido
                ref.getDownloadURL().subscribe(url => {
                    console.log("aqui obtiene el url de la canción", url)
                    
                    //envia parámetros album, url, referencia en FireStorage
                    this.albumCreate(object, url, this.filePath); 
                }  
                });
                });  
            }

Método para crear documento de álbum en Firestore 

            
            //variables genéricas para crear canción
            public genre_name: string;
            //función para crear un álbum 
            //necesita parámetros: objeto género, url, referencia en FireStorage
            albumCreate(album: Album, urlImg:any, filePath: any) {
                const id = this.angularFirestore.createId();//crea un ID
                //crea un documento con los campos especificados
                this.albumCollection
                .doc(id)//especifica el documento
                //establece y crea documento mediante los campos especificados
                .set({id,
                    //nombre del género por parte de variable general
                    genre_name: this.genre_name, 
                    name: album.name,
                    imageURL: urlImg, 
                    author: album.author, 
                    image_reference: filePath,
                    year: album.year });
                this.imageURL= urlImg;  
            }

Metodos en el Componente

Métodos para cargar y obtener los datos ingresados por el usario.

            
            import { SongService } from 'src/app/song.service'; //Servicio en donde estan los métodos genéricos
            import { FormBuilder, FormGroup } from '@angular/forms'; //Formulario reactivo
            import { ViewChild } from "@angular/core";

            public albumForm: FormGroup; //formulario reactivo
            public collectionName= "albumes" //nombre de colleción creada en firebase
            @ViewChild("file") file; //permite referenciar una instancia en HTML
            files: Set<File> = new Set(); 
            url: any =
                "https://i.pinimg.com/564x/65/df/2c/65df2c922e64c61235162ab7c0924d3c.jpg"; //imagen por defecto
            _file;
            
            constructor(
                public songService: SongService,
                public formBuilder: FormBuilder,
                public router: Router
            ) {
                //campos que maneja un álbum 
                this.albumForm= this.formBuilder.group({
                genre_name:[''],
                name: [''],
                author: [''],
                year:[''],
                image_reference: [''],
                imageURL: [''],
                id: null
                })
            }

            onSubmit(){
                //Referencia a función del servicio para crear album
                //envia como parámetros: valores del album reactivo, imagen local, nombre de la colección
                this.songService.addGenreAlbum(this.albumForm.value, this._file, this.collectionName);
                //Obtiene datos necesarios para la canción
                this.songService.getAlbumSongProperties(this.albumForm.value);
                //reseteo de file
                this.file.nativeElement.value = "";
            }

            //función para transformar la imagen subida a url local
            onFilesAdded(target: any) {
                this.isChanged = true;
                const reader = new FileReader(); //lector de archivos
                //función para cargar imagen subida y presentarla al usuario
                reader.onload = () => { 
                this.url = reader.result; 
                };

                if (target.files.length > 0) {
                this._file = target.files[0];
                //asignación de url local a _file
                reader.readAsDataURL(this._file); 
                }
            }
